#include <iostream>
using namespace std;

const int n = 3;           
int P[4];                  
int R[4];                  

void per(int k) {
    for (int i = 1; i <= n; i++) {
        if (R[i] == 0) {                
            P[k] = i;                    
            R[i] = 1;                   
            
            if (k == n) {                
                for (int j = 1; j <= n; j++)
                    cout << P[j] << " ";
                cout << endl;
            } else {
                per(k + 1);              
            }
            
            R[i] = 0;                    // возврат 
        }
    }
}

int main() {
    for (int i = 1; i <= n; i++)
        R[i] = 0;
    
    per(1);                              
    return 0;
}
